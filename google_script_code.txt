// كود Google Apps Script للرفع المباشر إلى GitHub
// انقل هذا الكود إلى مشروع جديد في https://script.google.com/

function doPost(e) {
  // 1. ضع التوكين الخاص بك هنا (GitHub Personal Access Token)
  // يجب أن يبدأ بـ ghp_ وله صلاحيات repo
  var GITHUB_TOKEN = "ضع_التوكين_الخاص_بك_هنا"; 
  
  var REPO_OWNER = "ahmedeid5";
  var REPO_NAME = "FB-Refiner-Community-Lists";

  try {
    // استقبال البيانات بطريقة الفورم (المتوافقة مع طريقة الموقع الجديدة)
    var fileNameOriginal = e.parameter.fileName;
    var content = e.parameter.content;
    
    if (!content || !fileNameOriginal) {
       throw new Error("Missing data: fileName or content");
    }

    // 2. تنظيف اسم الملف وإضافة تايم-ستامب لمنع التكرار
    var timestamp = Math.floor(Date.now() / 1000);
    var cleanName = fileNameOriginal.replace(/[^a-zA-Z0-9._-]/g, '_');
    var fileName = timestamp + "_" + cleanName;
    
    // 3. تحويل المحتوى لـ Base64 (شرط أساسي لـ GitHub API)
    var blob = Utilities.newBlob(content);
    var base64Content = Utilities.base64Encode(blob.getBytes());

    // 4. تجهيز الطلب لـ GitHub API (الرفع المباشر للمجلد)
    var url = "https://api.github.com/repos/" + REPO_OWNER + "/" + REPO_NAME + "/contents/community-lists/" + fileName;
    
    var payload = {
      "message": "Direct upload via Google Proxy: " + fileNameOriginal,
      "content": base64Content,
      "branch": "main"
    };

    var options = {
      "method": "put",
      "contentType": "application/json",
      "headers": { 
        "Authorization": "token " + GITHUB_TOKEN,
        "Accept": "application/vnd.github.v3+json"
      },
      "payload": JSON.stringify(payload),
      "muteHttpExceptions": true
    };

    var response = UrlFetchApp.fetch(url, options);
    var responseCode = response.getResponseCode();
    var responseBody = response.getContentText();
    
    if (responseCode == 201 || responseCode == 200) {
      // رد نجاح بصيغة HTML ليظهر للمستخدم في النافذة الجديدة
      return HtmlService.createHtmlOutput("<html><body style='font-family:sans-serif; text-align:center; padding-top:50px;'>" +
                                          "<h2>✅ تم الرفع بنجاح!</h2>" +
                                          "<p>تم إضافة الملف '" + fileNameOriginal + "' إلى المستودع.</p>" +
                                          "<p>يمكنك إغلاق هذه النافذة الآن.</p>" +
                                          "<script>setTimeout(window.close, 5000);</script></body></html>");
    } else {
      return HtmlService.createHtmlOutput("<h2>❌ حدث خطأ أثناء الرفع</h2><p>كود الخطأ: " + responseCode + "</p><pre>" + responseBody + "</pre>");
    }
    
  } catch (err) {
    return HtmlService.createHtmlOutput("<h2>❌ حدث خطأ تقني</h2><p>" + err.toString() + "</p>");
  }
}
